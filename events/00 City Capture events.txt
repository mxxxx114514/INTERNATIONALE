# Fall of London
news_event = {
	id = city_capture_ENG.1
	title = city_capture_ENG.1.t

	picture = GFX_news_event_capture_london

	major = yes
	is_triggered_only = yes

	immediate = {
		event_target:fallen_state = { set_state_flag = city_has_fallen_once }
	}

	option = {
		name = city_capture_ENG.1.a
		trigger = {
			OR = {
				tag = FROM.FROM
				has_war_together_with = FROM.FROM
			}
		}
	}

	option = {
		name = city_capture_ENG.1.b
		trigger = {
			OR = {
				tag = FROM
				has_war_together_with = FROM
			}
		}
	}

	option = {
		name = city_capture_ENG.1.c
		trigger = {
			NOT = {
				OR = {
					tag = FROM.FROM
					has_war_together_with = FROM.FROM
					tag = FROM
					has_war_together_with = FROM
				}
			}
		}
	}
}
