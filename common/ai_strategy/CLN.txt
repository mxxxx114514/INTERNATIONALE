	allowed = {
		OR = {
			tag = ENG
			tag = NFR
			tag = CAN
			tag = USF
			tag = RAJ
			tag = AST
			tag = POR
			tag = GRR
			tag = POL
			tag = BAL
			tag = LIT
			tag = WFI
		}

	}

	enable = {
		has_global_flag = fourth_balkan_war
		CZE = { OR = { has_war_with = HUN has_war_with = BUL has_war_with = YUG } }
	}

	abort = {
		OR = {

		}
	}

	ai_strategy = {
		type = send_volunteers_desire
		id = CZE
		value = 150
	}
	ai_strategy = {
		type = send_lend_lease_desire
		id = CZE
		value = 150
	}

}
