MEX_improve_catholic_opposition = {
	IF = {
		limit = { has_idea = MEX_catholic_opposition_4 }
		remove_ideas = MEX_catholic_opposition_4
	}
	ELSE_IF = {
		limit = { has_idea = MEX_catholic_opposition_3 }
		swap_ideas = {
			remove_idea = MEX_catholic_opposition_3
			add_idea = MEX_catholic_opposition_4
		}
	}
		ELSE_IF = {
		limit = { has_idea = MEX_catholic_opposition_2 }
		swap_ideas = {
			remove_idea = MEX_catholic_opposition_2
			add_idea = MEX_catholic_opposition_3
		}
	}
	ELSE_IF = {
		limit = { has_idea = MEX_catholic_opposition_1 }
		swap_ideas = {
			remove_idea = MEX_catholic_opposition_1
			add_idea = MEX_catholic_opposition_2
		}
	}
	ELSE_IF = {
		limit = { has_idea = MEX_catholic_opposition }
		swap_ideas = {
			remove_idea = MEX_catholic_opposition
			add_idea = MEX_catholic_opposition_1
		}
	}
	ELSE = {
		add_ideas = MEX_catholic_opposition
	}
}
